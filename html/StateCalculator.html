classdef StateCalculator <handle

    properties
        %状态变量
        V;%速度向量
        P;%位移向量
        Cbn;%姿态矩阵
        Phi;%欧拉角
        StateSeq;%状态向量序列

        %原始数据序列
        wSeq;%角速度序列
        fSeq;%加速度序列
        phiSeq;%欧拉角序列


        %常量
        delta_t;%更新间隔
        gravity;%重力加速度

        %核心组件
        gaitDetector;%步态检测器
        zuptKF;%卡尔曼滤波器（零速度修正特殊版本）



    end

    methods
       %构造函数
       function obj = StateCalculator()
            obj.delta_t = 0.006;%更新间隔 (秒)
            obj.gravity = 9.8;%重力加速度数值(米每平方秒)

            %创建步态检测器
            gaitDetector = GaitPhaseDetector();
            %创建卡尔曼滤波器
            zuptKF = ZuptKalmanFilter();

       end

       %状态初始化
       function stateInit(obj,phi0)
            phi0
            obj.Phi = phi0;
            obj.V = zeros(3,1);
            obj.P = zeros(3,1);
            obj.Cbn = getCbnFromPhi(phi0);
            %初始化状态序列结构体
            obj.StateSeq = struct('V',obj.V,'P',obj.P,'Cbn',obj.Cbn,'Phi',obj.Phi);
       end


       %将结果保存进序列
       function saveStateToSeq(obj)
           obj.StateSeq.V = [obj.StateSeq.V,obj.V];
           obj.StateSeq.P = [obj.StateSeq.P,obj.P];
           obj.StateSeq.Cbn = [obj.StateSeq.Cbn,obj.Cbn];
           obj.StateSeq.Phi = [obj.StateSeq.Phi,obj.Phi];
       end

       %状态解算
       function solveState(obj,phiSeq,wSeq,fSeq)
            %结果赋初值
            obj.wSeq = wSeq;
            obj.fSeq = fSeq;
            obj.phiSeq = phiSeq;
            obj.phiSeq(:,1)
            stateInit(obj,obj.phiSeq(:,1));%状态初始化

            %序列化迭代计算
            for i = 2:length(wSeq)
                %记录上一时刻状态量
                Cbn_prev = obj.Cbn;
                V_prev = obj.V;
                P_prev = obj.P;

                %计算当前时刻的状态量
                calculateNextCbn(Cbn_prev,wSeq(:,i));
                calculateNextV(V_prev,fSeq(:,i),obj.Cbn,fSeq(:,i-1),Cbn_prev);
                calculateNextP(P_prev,obj.V,V_prev);
                obj.Phi = getPhiFromCbn(obj.Cbn);

                %计算步态
                gaitPhase = obj.gaitDetector.getPhase(wSeq(:,i-obj.gaitDetector.windowLength:i),...
                                                      fSeq(:,i-obj.gaitDetector.windowLength:i));
                %依据步态执行零速度更新
                %零速度修正
                if gaitPhase == obj.gaitDetector.PhaseSwing
                     %摆动相
                    obj.zuptKF.X = zeros(9,1); %误差项置零
                elseif gaitPhase == obj.gaitDetector.PhaseStance
                    %支撑相
                    %预测
                    F = getF(obj.Cbn,fSeq(:,i));
                    G = getG(obj.Cbn);
                    obj.zuptKF.predict(F,G);
                    %更新
                    Z = -obj.V;
                    obj.zuptKF.update(Z);
                    %修正
                    reviseVn(obj.V);
                    revisePn(obj.P);
                    reviseCbn(obj.Cbn);
                    obj.Phi = getPhiFromCbn(obj.Cbn);

                 elseif  gaitPhase == obj.gaitDetector.PhaseUnknown %计算错误

                end

                %将计算结果保存
                saveStateToSeq();




            end

       end


       function Phi = getPhiFromCbn(Cbn)
            %横滚角Roll
            T33  = Cbn(3,3);
            T32 = Cbn(3,2);
            roll = atan(Cbn(3,2)/Cbn(3,3));
            %角度转换至[-pi,pi]
            if(T33>0)

            else
                if(T32>0)
                    roll =  pi + roll;
                else
                    roll =  -pi + roll;
                end
            end
            Phi(1) = roll;
            %俯仰角Pitch
            Phi(2) = asin(-Cbn(3,1));
            %航向角yaw
            yaw = atan(Cbn(2,1)/Cbn(1,1));
            T11 = Cbn(1,1);
            T21 = Cbn(2,1);
            %角度转换至[-pi,pi]
            if(T11<0&&T21<0)
                yaw = yaw - pi;
            end
            if(T11<0&&T21>0)
                yaw =  yaw + pi;
            end
            Phi(3) = yaw;

            Phi = Phi'*180/pi;%转置成列向量
        end

       %从欧拉角获取姿态矩阵
       function  Cbn = getCbnFromPhi1(phi)
            phi =  phi *pi/180;%弧度制转角度值

            gamma = phi(1);%横滚角roll
            theta = phi(2);%俯仰角pitch
            psi = phi(3);%翻滚角yaw

            Cbn = [cos(theta)*cos(psi),-cos(gamma)*sin(psi)+sin(gamma)*sin(theta)*cos(psi),sin(gamma)*sin(psi)+cos(gamma)*sin(theta)*cos(psi);
                   cos(theta)*sin(psi),cos(gamma)*cos(psi)+sin(gamma)*sin(theta)*sin(psi),-sin(gamma)*cos(psi)+cos(gamma)*sin(theta)*sin(psi);
                   -sin(theta),sin(gamma)*cos(theta),cos(gamma)*cos(theta)];
        end

        %姿态矩阵计算——二子样算法
        function  calculateNextCbn(obj,w)
            w = w * pi /180;%弧度制转角度制
            I = eye(3);
            wx = getCrossMatrix(w);%角速度w构成的反对称矩阵[w×]
            obj.Cbn = Cbn_prev * ((2*I + wx*DELTA_T)/(2*I - wx*DELTA_T));
        end

        %计算下一时刻的速度
        function calculateNextV(obj,Vn_prev,fn,Cbn,fn_prev,Cbn_prev)
            obj.V = Vn_prev + (0.5*(Cbn*fn+Cbn_prev*fn_prev) - GRAVITY ) * DELTA_T;
        end

        %计算下一时刻的位移
        function calculateNextP(obj,Pn_prev,Vn,Vn_prev)
            obj.P = Pn_prev + 0.5*(Vn_prev + Vn) * obj.delta_t;
        end

        %获取状态转移矩阵F
        function F = getF(Cbn,fn,delta_t)
            I = eye(3);
            ZeroMatrix = zeros(3,3);
            F = [I,ZeroMatrix,ZeroMatrix;
                 (getCrossMatrix(Cbn*fn))*delta_t,I,ZeroMatrix;
                 ZeroMatrix,I*delta_t,I];
        end

        %获取误差分配矩阵G
        function G = getG(Cbn,delta_t)
            ZeroMatrix = zeros(3,3);
            G = [-Cbn*delta_t,ZeroMatrix;
                 ZeroMatrix,Cbn*delta_t;
                 ZeroMatrix,ZeroMatrix];
        end

       %速度更新
       function reviseVn(obj,vel_corr)
            obj.V = obj.V + vel_error;
       end

        %位移更新
        function revisePn(obj,pos_error)
            obj.P = obj.P + pos_error;
        end

        %姿态更新
        function reviseCbn(obj,phi_error)
            I = eye(3);
            phi_error_x = getCrossMatrix(phi_error)*180/pi;
            obj.Cbn =  ((2*I + phi_error_x)\(2*I - phi_error_x)) * obj.Cbn;
        end


        % 计算三维向量的反对称矩阵
        function crossMatrix = getCrossMatrix(v)
            crossMatrix = [0, -v(3), v(2);
                           v(3), 0, -v(1);
                           -v(2), v(1), 0];
        end



    end
end
